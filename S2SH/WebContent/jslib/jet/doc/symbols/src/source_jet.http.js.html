 36 	// 兼容不同浏览器的 Adapter 适配层
 37 	if(typeof window.XMLHttpRequest === "undefined"){
 38 		window.XMLHttpRequest = function(){
 39 			return new window.ActiveXObject(navigator.userAgent.indexOf("MSIE 5") >=0 ? "Microsoft.XMLHTTP" : "Msxml2.XMLHTTP");
 40 		};

 51 	/**
 52 	 * 这是Ajax对象名字空间的一个方法
 53 	 * 
 54 	 * @memberOf http
 55 	 * @method	ajax
 56 	 * 
 57 	 * @param {Object} options 一个配置对象
 58 	 * @return {Object} ajax 返回一个ajax对象
 59 	 */
 60 	ajax = function(uri, options){
 61 		var httpRequest,
 62 			httpSuccess,
 63 			timeout,
 64 			isTimeout = false,
 65 			isComplete = false;
 66 		
 67 		options = {
 68 			method: options.method || "GET",
 69 			data: options.data || null,
 70 			arguments: options.arguments || null,
 71 
 72 			onSuccess: options.onSuccess || function(){},
 73 			onError: options.onError || function(){},
 74 			onComplete: options.onComplete || function(){},
 75 			//尚未测试
 76 			onTimeout: options.onTimeout || function(){},
 77 
 78 			isAsync: options.isAsync || true,
 79 			timeout: options.timeout ? options.timeout : 10000,
 80 			contentType: options.contentType ? options.contentType : "utf-8",
 81 			type: options.type || "xml"
 82 		};
 83 		uri = uri || "";
 84 		timeout = options.timeout;
 85 		
 86 		
 87 		httpRequest = new window.XMLHttpRequest();
 88 		httpRequest.open(options.method, uri, options.isAsync);
 89 		//设置编码集
 90 		//httpRequest.setRequestHeader('Content-Type','application/x-www-form-urlencoded');
 91 		httpRequest.setRequestHeader("Content-Type",options.contentType);
 92 
 93 		/**
 94 		 * @ignore
 95 		 */
 96 		httpSuccess = function(r){
 97 			try{
 98 				return (!r.status && location.protocol == "file:")
 99 					|| (r.status>=200 && r.status<300)
100 					|| (r.status==304)
101 					|| (navigator.userAgent.indexOf("Safari")>-1 && typeof r.status=="undefined");
102 			}catch(e){
103 				//J.out("错误：[" + e.name + "] "+e.message+", " + e.fileName+", 行号:"+e.lineNumber+"; stack:"+typeof e.stack, 2);
104 			}
105 			return false;
106 		}
107 		
108 
109 		httpRequest.onreadystatechange=function (){
110 			if(httpRequest.readyState==4){
111 				if(!isTimeout){
112 					var o={};
113 						o.responseText = httpRequest.responseText;
114 						o.responseXML = httpRequest.responseXML;
115 						o.uri=uri;
116 						o.arguments=options.arguments;
117 						
118 					if(httpSuccess(httpRequest)){
119 						if(options.type === "script"){
120 							eval.call(window, data);
121 						}
122 						options.onSuccess(o);
123 						
124 					}else{
125 						options.onError(o);
126 					}
127 					options.onComplete(o);
128 				}
129 				isComplete = true;
130 				//删除对象,防止内存溢出
131 				httpRequest=null;
132 			}
133 		};
134 		
135 		httpRequest.send(options.data);
136 		
137 		window.setTimeout(function(){
138 			var o;
139 			if(!isComplete){
140 				isTimeout = true;
141 				o={};
142 				o.uri=uri;
143 				o.arguments=options.arguments;
144 				options.onTimeout(o);
145 				options.onComplete(o);
146 			}
147 		}, timeout);	
148 		
149 		return httpRequest;
150 	};
151 
152 	
153 	/**
154 	 * comet方法
155 	 * 
156 	 * @memberOf http
157 	 * @method	comet
158 	 * @param {String} uri uri地址
159 	 * @param {Object} options 配置对象
160 	 * @return {Object} 返回一个comet dom对象
161 	 */
162 	comet = function(uri, options){
163 
164 		uri = uri || "";
165 		options = {
166 			method : options.method || "GET",
167 			data : options.data || null,
168 			arguments : options.arguments || null,
169 			callback : options.callback || function(){},
170 			onLoad : options.onLoad || function(){},
171 
172 			contentType: options.contentType ? options.contentType : "utf-8"
173 		};
174 
175 		var connection;
176 		if(J.browser.ie){
177 			var htmlfile = new ActiveXObject("htmlfile");
178 			htmlfile.open();
179 			htmlfile.close();
180 			var iframediv = htmlfile.createElement("div");
181 			htmlfile.appendChild(iframediv);
182 			htmlfile.parentWindow._parent = self;
183       		iframediv.innerHTML = '<iframe id="_cometIframe" src="'+uri+'?callback=window.parent._parent.'+options.callback+'"></iframe>';
184       		
185 			connection = htmlfile.getElementById("_cometIframe");
186 		
187 		}
188 		else{
189 			connection = $D.node("iframe");
190 			connection.setAttribute("id", "_cometIframe");
191 			connection.setAttribute("src", uri+'?callback=window.parent._parent.'+options.callback);
192 			connection.style.position = "absolute";
193 			connection.style.visibility = "hidden";
194 			connection.style.left = connection.style.top = "-999px";
195 			connection.style.width = connection.style.height = "1px";
196 			document.body.appendChild(connection);
197 			self._parent = self;
198 		};
199 
200 		$E.on(connection,"load", options.onLoad);
201 
202 		return connection;
203 		
204 	};

211 	/**
212 	 * 这是load方法
213 	 * 
214 	 * @memberOf http
215 	 * @method load
216 	 * 
217 	 * @param {String} type 一个配置对象
218 	 * @param {Object} options 一个配置对象
219 	 * @return {Object} ajax 返回一个ajax对象
220 	 */
221 	load = function(type, uri, options){
222 		var node,
223 			linkNode,
224 			scriptNode,
225 			id,
226 			head = document.getElementsByTagName("head") ? document.getElementsByTagName("head")[0] : document.documentElement,
227 			timer,
228 			isTimeout = false,
229 			isComplete = false,
230 			options = options || {},
231 			query = options.query || null,
232 			arguments = options.arguments || null,
233 			
234 			onSuccess = options.onSuccess || function(){},
235 			onError = options.onError || function(){},
236 			onComplete = options.onComplete || function(){},
237 			purge,
238 			//尚未测试
239 			onTimeout = options.onTimeout || function(){},
240 
241 			timeout = options.timeout ? options.timeout : 10000,
242 			charset = options.charset ? options.charset : "utf-8",
243 			win = options.win || window,
244 			o,
245 			
246 			getId;
247 
248         uri = uri || "";
249 		if(query !== null){
250 			uri = uri + "?" + query;
251 		}
252 		/**
253 		 * @ignore
254 		 */
255 		getId = function(){
256 	    	return load.Id++;
257 	    };
258 	    id = getId();
259 	    
260 	    /**
261 		 * @ignore
262 		 */
263 	    purge = function(id){
264 	    	head.removeChild($("jet_load_" + id));
265 	    };
266 
267         /**
268 	     * Generates a link node
269 	     * @method _linkNode
270 	     * @param uri {string} the uri for the css file
271 	     * @param win {Window} optional window to create the node in
272 	     * @return {HTMLElement} the generated node
273 	     * @private
274 	     */
275 	    linkNode = function(uri, win, charset) {
276 	        var c = charset || "utf-8";
277 	        return $D.node("link", {
278 		                "id":		"jet_load_" + id,
279 		                "type":		"text/css",
280 		                "charset":	c,
281 		                "rel":		"stylesheet",
282 		                "href":		uri
283 		            }, win);
284 	    };
285 	    
286 		/**
287 	     * Generates a script node
288 	     * @method _scriptNode
289 	     * @param uri {string} the uri for the script file
290 	     * @param win {Window} optional window to create the node in
291 	     * @return {HTMLElement} the generated node
292 	     * @private
293 	     */
294 	    scriptNode = function(uri, win, charset) {
295 	        var c = charset || "utf-8";
296 	        return $D.node("script", {
297 		                "id":		"jet_load_" + id,
298 		                "defer":	"defer",
299 		                "type":		"text/javascript",
300 		                "charset":	c,
301 		                "src":		uri
302 		            }, win);
303 	    };
304 	    
305         
306 	    
307 	    if(type === "script"){
308             node = options.node || scriptNode(uri, win, charset);
309         }else if(type === "css"){
310             node = options.node || linkNode(uri, win, charset);
311         }
312         
313         
314         
315         if(J.browser.engine.trident){
316             node.onreadystatechange = function() {
317                 var rs = this.readyState;
318                 if (rs === "loaded" || rs === "complete") {
319                     node.onreadystatechange = null;
320 
321                     if(!isTimeout){
322                     	isComplete = true;
323                     	window.clearTimeout(timer);
324                 		timer = null;
325                     	o={};
326 						o.id = id;
327 						o.uri = uri;
328 						o.arguments = arguments;
329                     	onSuccess(o);
330                     	onComplete(o);
331                     	if(type === "script"){
332 	                		purge(id);
333 	                	}
334                     }
335                 }
336             };
337 
338         // webkit prior to 3.x is no longer supported
339         }else if(J.browser.engine.webkit){
340 			
341             // Safari 3.x supports the load event for script nodes (DOM2)
342             $E.on(node, "load", function(){
343             	var o;
344                 if(!isTimeout){
345                 	isComplete = true;
346                 	window.clearTimeout(timer);
347                 	timer = null;
348                 	o={};
349 					o.id = id;
350 					o.uri = uri;
351 					o.arguments = arguments;
352                 	onSuccess(o);
353                 	onComplete(o);
354                 	if(type === "script"){
355                 		purge(id);
356                 	}
357                 }
358             });
359 
360 
361         // FireFox and Opera support onload (but not DOM2 in FF) handlers for
362         // script nodes.  Opera, but not FF, supports the onload event for link
363         // nodes.
364         }else{ 
365 			
366             node.onload = function(){
367             	var o;
368             	//J.out("else:"+J.browser.engine.name);
369                 if(!isTimeout){
370                 	isComplete = true;
371                 	window.clearTimeout(timer);
372                 	timer = null;
373                 	o={};
374 					o.id = id;
375 					o.uri = uri;
376 					o.arguments = options.arguments;
377                 	onSuccess(o);
378                 	onComplete(o);
379                 	
380                 	if(type === "script"){
381                 		purge(id);
382                 	}
383                 }
384             };
385 
386             node.onerror = function(e){
387             	var o;
388             	//J.out("else:"+J.browser.engine.name);
389                 if(!isTimeout){
390                 	isComplete = true;
391                 	window.clearTimeout(timer);
392                 	timer = null;
393                 	o={};
394 					o.id = id;
395 					o.uri = uri;
396 					o.arguments = arguments;
397 					o.error = e;
398                 	onError(o);
399                 	onComplete(o);
400                 	//if(type === "script"){
401                 		purge(id);
402                 	//}
403                 }
404             };
405         }
406         if(options.node){
407         	
408         	if(type === "script"){
409 	            node.src = uri;
410 	        }else if(type === "css"){
411 	            node.href = uri;
412 	        }
413         }else{
414         	head.appendChild(node);
415         }
416        
417         
418         if(type === "script"){
419             timer = window.setTimeout(function(){
420 				var o;
421 				if(!isComplete){
422 					isTimeout = true;
423 					o = {};
424 					o.uri = uri;
425 					o.arguments = arguments;
426 					onTimeout(o);
427 					onComplete(o);
428 					purge(id);
429 				}
430 			}, timeout);
431         }
432         
433 		var func = function(node){
434 			this._node = node;
435 			this._head = head;
436 		};
437 		func.prototype={
438 			abort:function(){
439 				this._node.src="";
440 				this._head.removeChild(this._node);
441 				delete this._node;
442 			}
443 			
444 		};
445 		return new func(node);
446 	};
447 	load.Id=0;
448 	
449 	/**
450 	 * 加载CSS
451 	 * 
452 	 * @memberOf http
453 	 * @method loadCss
454 	 * 
455 	 * @param {String} uri 一个uri
456 	 * @param {Object} options 一个配置对象
457 	 * @return {Object} ajax 返回一个ajax对象
458 	 */
459 	loadCss = function(uri, options){
460 		return load("css", uri, options);
461 	};
462 	
463 	/**
464 	 * 加载Javascript
465 	 * 
466 	 * @memberOf http
467 	 * @method loadScript
468 	 * 
469 	 * @param {String} uri 一个uri
470 	 * @param {Object} options 一个配置对象
471 	 * @return {Object} ajax 返回一个ajax对象
472 	 */
473 	loadScript = function(uri, options){
474 		return load("script", uri, options);
475 	};
476 	
477 	J.http.ajax = ajax;
478 	J.http.comet = comet;
479 	J.http.load = load;
480 	J.http.loadCss = loadCss;
481 	J.http.loadScript = loadScript;
482 });
