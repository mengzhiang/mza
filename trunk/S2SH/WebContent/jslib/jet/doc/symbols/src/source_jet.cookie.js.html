 24 /**
 25  * cookie类
 26  * 
 27  * @namespace J.cookie
 28  */
 29 Jet().$package(function(J){
 30 	var domainPrefix = window.location.host;
 31 	
 32 	/**
 33 	 * @namespace cookie 名字空间
 34 	 * @name cookie
 35 	 */
 36 	J.cookie = 
 37 	/**
 38 	 * @lends cookie
 39 	 */	
 40 	{
 41 		
 42 		/**
 43 		 * 设置一个cookie
 44 		 * 
 45 		 * @param {String} name cookie名称
 46 		 * @param {String} value cookie值
 47 		 * @param {String} domain 所在域名
 48 		 * @param {String} path 所在路径
 49 		 * @param {Number} hour 存活时间，单位:小时
 50 		 * @return {Boolean} 是否成功
 51 		 */
 52 		set : function(name, value, domain, path, hour) {
 53 			if (hour) {
 54 				var today = new Date();
 55 				var expire = new Date();
 56 				expire.setTime(today.getTime() + 3600000 * hour);
 57 			}
 58 			window.document.cookie = name + "=" + value + "; " + (hour ? ("expires=" + expire.toGMTString() + "; ") : "") + (path ? ("path=" + path + "; ") : "path=/; ") + (domain ? ("domain=" + domain + ";") : ("domain=" + domainPrefix + ";"));
 59 			return true;
 60 		},
 61 	
 62 		/**
 63 		 * 获取指定名称的cookie值
 64 		 * 
 65 		 * @param {String} name cookie名称
 66 		 * @return {String} 获取到的cookie值
 67 		 */
 68 		get : function(name) {
 69 			var r = new RegExp("(?:^|;+|\\s+)" + name + "=([^;]*)");
 70 			// var r = new RegExp("(?:^|;+|\\s+)" + name + "=([^;]*?)(?:;|$)");
 71 			var m = window.document.cookie.match(r);
 72 			return (!m ? "" : m[1]);
 73 			// document.cookie.match(new
 74 			// RegExp("(?:^|;+|\\s+)speedMode=([^;]*?)(?:;|$)"))
 75 		},
 76 	
 77 		/**
 78 		 * 删除指定cookie,复写为过期
 79 		 * 
 80 		 * @param {String} name cookie名称
 81 		 * @param {String} domain 所在域
 82 		 * @param {String} path 所在路径
 83 		 */
 84 		remove : function(name, domain, path) {
 85 			window.document.cookie = name + "=; expires=Mon, 26 Jul 1997 05:00:00 GMT; " + (path ? ("path=" + path + "; ") : "path=/; ") + (domain ? ("domain=" + domain + ";") : ("domain=" + domainPrefix + ";"));
 86 		}
 87 	};
 88 
 89 });
