 40  * 拖拽模块
 41  */
 42 Jet().$package(function(J){
 43 	var $D = J.dom,
 44 		$E = J.event;
 45 
 46 	J.ui.Drag = new J.Class({
 47 		init:function(apperceiveEl, effectEl){
 48 			var context = this;
 49 			var curDragElementX, curDragElementY, dragStartX, dragStartY;
 50 			this.apperceiveEl = apperceiveEl;
 51 			if(effectEl === false){
 52 				this.effectEl = false;
 53 			}else{
 54 				this.effectEl = effectEl || apperceiveEl;
 55 			}
 56 			
 57 			
 58 			var ieSelectFix = function(e) {
 59 				e.preventDefault();
 60 	            //return false;
 61 	        };
 62 	        
 63 			this.dragStart = function(e){
 64 				e.preventDefault();
 65 				e.stopPropagation();
 66 				curDragElementX = parseInt($D.getStyle(context.effectEl,"left")) || 0;
 67 				curDragElementY = parseInt($D.getStyle(context.effectEl,"top")) || 0;
 68 				dragStartX = e.pageX;
 69 				dragStartY = e.pageY;
 70 				$E.on(document, "mousemove", context.dragMove);
 71 				$E.on(document, "mouseup", context.dragStop);
 72 				if(J.browser.ie){
 73 					$E.on(document.body, "selectstart", ieSelectFix);
 74 				}
 75 				
 76 				
 77 				$E.notifyObservers(context, "start", {x:curDragElementX, y:curDragElementY});
 78 			};
 79 	
 80 			this.dragMove = function(e){
 81 
 82 				var x = curDragElementX+(e.pageX-dragStartX);
 83 				var y = curDragElementY+(e.pageY-dragStartY);
 84 				var isMoved = false;
 85 				
 86 				if(context._oldX !== x){
 87 					context._oldX = x;
 88 					if(context.effectEl){
 89 						context.effectEl.style.left = x+"px";
 90 					}
 91 					isMoved = true;
 92 				}
 93 				if(context._oldY !== y){
 94 					context._oldY = y;
 95 					if(context.effectEl){
 96 						context.effectEl.style.top = y+"px";
 97 					}
 98 					isMoved = true;
 99 				}
100 				if(isMoved){
101 					$E.notifyObservers(context, "move", {x:x, y:y});
102 				}
103 				
104 			};
105 	
106 			this.dragStop = function(e){
107 				$E.off(document, "mousemove", context.dragMove);
108 				$E.off(document, "mouseup", context.dragStop);
109 				if(J.browser.ie){
110 					$E.off(document.body, "selectstart", ieSelectFix);
111 				}
112 				$E.notifyObservers(context, "finish", {x:context._oldX, y:context._oldY});
113 			};
114 			
115 			$E.on(this.apperceiveEl, "mousedown", this.dragStart);
116 		},
117 		lock : function() {
118 			$E.off(this.apperceiveEl, "mousedown", this.dragStart);
119 		},
120 		unlock : function(){
121 			$E.on(this.apperceiveEl, "mousedown", this.dragStart);
122 		},
123 		show : function(){
124 			$D.show(this.apperceiveEl);
125 		},
126 		hide : function(){
127 			$D.hide(this.apperceiveEl);
128 		}
129 	});
130

402 /**
403  * tab模块
404  */
405 Jet().$package(function(J){
406 	var $ = J.dom.id,
407 		$D = J.dom,
408 		$E = J.event;
409 		
410 	/**
411 	 * Tab 类
412 	 * 
413 	 * @class
414 	 * @name Tab
415 	 */
416 	J.ui.Tab = function(triggers,sheets,config){
417 		this.tabs = [];             //tab的集合
418 		this.currentTab = null;     //当前tab
419 		this.config = {
420 			defaultIndex : 0,       //默认的tab索引
421 			triggerEvent : 'click', //默认的触发事件
422 			slideEnabled : false,   //是否自动切换
423 			slideInterval : 5*1000,   //切换时间间隔
424 			slideDelay : 300,       //鼠标离开tab继续切换的延时
425 			autoInit : true,        //是否自动初始化
426 			onShow:function(){ }    //默认的onShow事件处理函数
427 		};
428 	
429 		this.setConfig(config);
430 
431 		if(triggers && sheets) {
432 			this.addRange(triggers,sheets);
433 			if(this.config.autoInit){
434 				this.init();
435 			}
436 		}
437 	};
438 	
439 	J.ui.Tab.prototype = {
440 		/**
441 		 * 设置config
442 		 * @param {object} config 配置项如{'slideEnabled':true,'defaultIndex':0,'autoInit':false}
443 		 */
444 		setConfig:function(config){
445 			if(!config) return;
446 			for(var i in config){
447 				this.config[i] = config[i];
448 			}
449 		},
450 		/**
451 		 * 增加tab
452 		 * @param tab={trigger:aaaa, sheet:bbbb} 
453 		 * @param {string|HTMLElement} trigger
454 		 * @param {string|HTMLElement} sheet
455 		 * */
456 		add:function(tab){
457 			if(!tab) return;
458 			
459 			if(tab.trigger){
460 				this.tabs.push(tab);
461 				tab.trigger.style.display = 'block';
462 			}
463 		},
464 		/**
465 		 * 增加tab数组
466 		 * @param {array} triggers HTMLElement数组
467 		 * @param {array} sheets HTMLElement数组
468 		 * */
469 		addRange:function(triggers, sheets){
470 			if(!triggers||!sheets) return;
471 			if(triggers.length && sheets.length && triggers.length == sheets.length){
472 				for(var i = 0, len = triggers.length; i < len; i++){
473 					this.add({trigger:triggers[i],sheet:sheets[i]});
474 				}
475 			}
476 		},
477 		/**
478 		 * 设置tab为当前tab并显示
479 		 * @param {object} tab  tab对象 {trigger:HTMLElement,sheet:HTMLElement}
480 		 * */
481 		select:function(tab){
482 			if(!tab || (!!this.currentTab && tab.trigger == this.currentTab.trigger)) return;
483 			if(this.currentTab){
484 				$D.removeClass(this.currentTab.trigger, 'current');
485 				if(this.currentTab.sheet){
486 					this.currentTab.sheet.style.display = 'none';
487 				}
488 				
489 			}
490 			this.currentTab = tab;
491 			this.show();
492 		},
493 		
494 		/**
495 		 * 设置tab为隐藏的
496 		 * @param {object} tab  tab对象 {trigger:HTMLElement,sheet:HTMLElement}
497 		 * */
498 		remove:function(tab){
499 			if(!tab) return;
500 			
501 			
502 			if(tab.trigger){
503 				$D.removeClass(tab.trigger, 'current');
504 				tab.trigger.style.display = 'none';
505 			}
506 			if(tab.sheet){
507 				tab.sheet.style.display = 'none';
508 			}
509 			
510 			var index=this.indexOf(tab);
511 			this.tabs.splice(index,index);
512 	
513 			if(tab.trigger == this.currentTab.trigger){
514 				if(index==0){
515 					//this.currentTab=this.tabs[(index+1)];
516 					this.select(this.tabs[(index+1)]);
517 				}else{
518 					//this.currentTab=this.tabs[(index-1)];
519 					this.select(this.tabs[(index-1)]);
520 				}
521 			}
522 		},
523 		/**
524 		 * 显示当前被选中的tab
525 		 * */
526 		show:function(){
527 			
528 			if(this.currentTab.trigger){
529 				this.currentTab.trigger.style.display = 'block';
530 			}
531 			$D.addClass(this.currentTab.trigger, 'current');
532 			if(this.currentTab.sheet){
533 				this.currentTab.sheet.style.display = 'block';
534 			}
535 			$E.notifyObservers(this, "show", this.currentTab);
536 
537 		},
538 		/**
539 		 * 自动切换
540 		 * */
541 		slide:function(){
542 			var	config = this.config,
543 				_this = this,
544 				intervalId,
545 				delayId;
546 			J.array.forEach(this.tabs, function(tab, index, tabs){
547 				$E.on(tab.trigger, 'mouseover' , clear);
548 				$E.on(tab.sheet, 'mouseover' , clear);
549 				
550 				$E.on(tab.trigger, 'mouseout' , delay);
551 				$E.on(tab.sheet, 'mouseout' , delay);
552 			});
553 			start();
554 			function start() {
555 				var i = _this.indexOf(_this.currentTab);
556 				if( i == -1 ) return;
557 				intervalId = window.setInterval(function(){
558 					var tab = _this.tabs[ ++i % _this.tabs.length ];
559 					if(tab){
560 						_this.select(tab);
561 					}
562 				},config['slideInterval']);
563 			}
564 			function clear() {
565 				window.clearTimeout(delayId);
566 				window.clearInterval(intervalId);	
567 			}
568 			function delay() {
569 				delayId = window.setTimeout(start,config['slideDelay']);
570 			}
571 		},
572 		/**
573 		 * 获取tab在tabs数组中的索引
574 		 * @param {object} tab  tab对象 {trigger:HTMLElement,sheet:HTMLElement}
575 		 * */
576 		indexOf:function(tab){
577 			for(var i = 0, len = this.tabs.length; i < len; i++){
578 				if(tab.trigger == this.tabs[i].trigger)
579 					return i;
580 			}
581 			return -1;
582 		},
583 		/**
584 		 * 初始化函数
585 		 * */
586 		init:function(){
587 			var config = this.config,
588 				_this = this;
589 
590 			J.array.forEach(this.tabs, function(tab, index, tabs){
591 				$E.on(tab.trigger,config['triggerEvent'], function(){
592 					_this.select.call(_this,tab);
593 				});
594 				if(tab.sheet){
595 					tab.sheet.style.display = 'none';
596 				}
597 			});
598 			
599 			this.select(this.tabs[config['defaultIndex']]);
600 			if(config['slideEnabled']){
601 				this.slide();
602 			}
603 		}
604 	};
605 
606 });

636
