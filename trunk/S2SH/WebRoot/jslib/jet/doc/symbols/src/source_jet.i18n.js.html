 14 /**	
 15  * @description
 16  * Package: jet.i18n
 17  *
 18  * Need package:
 19  * jet.core.js
 20  * 
 21  */
 22  
 23 /**
 24  * 5.[Javascript core]: i18n 国际化扩展
 25  */
 26 Jet().$package(function(J){
 27 	
 28 	/**
 29 	 * i18n 名字空间
 30 	 * 
 31 	 * @namespace
 32 	 * @name i18n
 33 	 */
 34 	J.i18n = J.i18n || {};
 35 	
 36 	J.extend(J.i18n, 
 37 		
 38 	/**
 39 	 * @lends i18n
 40 	 * 
 41 	 */		
 42 	{
 43 		
 44 		/**
 45 		 * 引用所有语言包的对象
 46 		 * 
 47 		 * @memberOf i18n
 48 		 * @name langs
 49 		 * @type {Object}
 50 		 * 
 51 		 */
 52 		langs: {
 53 			"en-us": {
 54 				number:"ddd,ddd.dd",
 55 				currency:"$ddd,ddd.dd",
 56 				date: "MM/dd/yyyy",
 57 				longDate:"yyyy year MM month dd day",
 58 				time: "hh:mm:ss",
 59 				text: {
 60 					helloWorld:"Hello Word!",
 61 					test: "test",
 62 					about: "Jet Lib, Developer: Jetyu!"
 63 				}
 64 				
 65 			}
 66 		},
 67 
 68 		/**
 69 		 * 当前语言
 70 		 * 
 71     	 * @memberOf i18n
 72 		 * @name currentLang
 73 		 * @type {Object}
 74 		 */
 75 		currentLang: null,
 76 		
 77 		/**
 78 		 * 添加语言包对象
 79 		 * 
 80 		 * @param {Object} langObj 要添加的语言包对象
 81 		 * 
 82 		 * @example
 83 		 * Jet().$package(function(J){
 84 		 * 	// 添加一个名字为"zh-cn"中文语言包
 85 		 * 	J.i18n.add({
 86 		 * 		"zh-cn":{
 87 		 * 			// 数字
 88 		 * 			number:"ddd,ddd.dd",
 89 		 * 			// 货币
 90 		 * 			currency:"￥ddd,ddd.dd元",
 91 		 * 			// 日期
 92 		 * 			date: "yyyy-MM-dd",
 93 		 * 			// 长日期
 94 		 * 			longDate:"yyyy年MM月dd日",
 95 		 * 			// 时间
 96 		 * 			time: "hh小时mm分ss秒",
 97 		 * 			// 文本
 98 		 * 			text: {
 99 		 * 				helloWorld:"你好,世界!",
100 		 * 				test: "测试",
101 		 * 				onlyDefaultHas: "只有默认语言才有的...",
102 		 * 				about: "Jet 框架, 开发者: 于涛!"
103 		 * 			}	
104 		 * 		}
105 		 * 	});
106 		 * };
107 		 */
108 		add: function(langObj){
109 			return J.extend(J.i18n.langs, langObj);
110 		},
111 		
112 		/**
113     	 * 设置默认语言
114 		 * 
115 		 * @param {String} lang 语言包的名称
116 		 * 
117 		 * @example
118 		 * Jet().$package(function(J){
119 		 * 	// 将名字为"zh-cn"语言包设置为默认
120 		 * 	J.i18n.setDefault("zh-cn");
121 		 * };
122 		 */
123 		setDefault: function(lang){
124 			this.defaultLang = J.i18n.langs[lang];
125 			return this.defaultLang;
126 		},
127 		
128 		/**
129     	 * 设置为当前语言
130 		 * 
131 		 * @param {String} lang 语言包的名称
132 		 * @param {Boolean} isSetDefault 是否同时设置为默认的语言
133 		 * 
134 		 * @example
135 		 * Jet().$package(function(J){
136 		 * 	// 将名字为"zh-cn"语言包设置为当前语言
137 		 * 	J.i18n.set("zh-cn");
138 		 * };
139 		 */
140 		set: function(lang, isSetDefault){
141 			this.currentLang = J.i18n.langs[lang];
142 			if(!this.defaultLang){
143 				this.defaultLang = this.currentLang;
144 			}
145 			this.text = this.text || {};
146 			return J.extend(this.text, this.defaultLang, this.currentLang.text);
147 		},
148 		
149 		/**
150     	 * 按当前语言输出数字
151     	 * 
152 		 * @param {Mixed} 数字
153 		 * @returns {String} 返回按当前语言输出的数字的字符串形式
154 		 */
155 		number:function(s){
156 
157 	        if(/[^0-9\.]/.test(s)){
158 	        	return "invalid value";
159 	        }
160 	        s = s.replace(/^(\d*)$/,"$1.");
161 	        s = (s+"00").replace(/(\d*\.\d\d)\d*/,"$1");
162 	        s = s.replace(".",",");
163 	        var re = /(\d)(\d{3},)/;
164 	        while(re.test(s)){
165 	        	s = s.replace(re,"$1,$2");
166 	        }
167 	        s = s.replace(/,(\d\d)$/,".$1");
168 	        return s.replace(/^\./,"0.");
169 		},
170 		
171 		/**
172     	 * 按当前语言输出日期
173     	 * 
174 		 * @param {Mixed} 日期
175 		 * @returns {String} 返回按当前语言输出的日期字符串
176 		 * 
177 		 * @example
178 		 * Jet().$package(function(J){
179 		 * var d = new Date();
180 		 * 	// 安当前语言的格式输出时间字符串
181 		 * 	J.out(J.i18n.date(d));
182 		 * };
183 		 * 
184 		 */
185 		date: function(date){
186 			return J.formatDate(date, J.i18n.currentLang.date);
187 		},
188 		
189 		/**
190     	 * 按当前语言输出长日期
191     	 * 
192 		 * @param {Mixed} 长日期
193 		 * @returns {String} 返回按当前语言输出的长日期字符串
194 		 */
195 		longDate: function(date){
196 			return J.formatDate(date, J.i18n.currentLang.longDate);
197 		},
198 		
199 		/**
200     	 * 按当前语言输出时间
201     	 * 
202 		 * @param {Mixed} 时间
203 		 * @returns {String} 返回按当前语言输出的时间字符串
204 		 */
205 		time: function(time){
206 			return J.formatDate(date, J.i18n.currentLang.time);
207 		},
208 		
209 		/**
210     	 * @ignore
211 		 * @lends i18n
212 		 */
213 		toString: function(){
214 			return "";
215 		}
216 	
217 	
218 	});
219 	
220 	//将第一个设为默认
221 	for(var p in J.i18n.langs){
222 		J.i18n.set(p);
223 		break;
224 	}
225 
226 });
